    <script>

        window.onload = function() { loadHistory(); };

 

        function calculate() {

            const name = document.getElementById("cellName").value || "Unknown";

            const sq1 = parseFloat(document.getElementById("sq1").value) || 0;

            const sq2 = parseFloat(document.getElementById("sq2").value) || 0;

            const sq3 = parseFloat(document.getElementById("sq3").value) || 0;

            const sq4 = parseFloat(document.getElementById("sq4").value) || 0;

            const df = parseFloat(document.getElementById("dilutionFactor").value) || 20;

            const V_init = parseFloat(document.getElementById("volInit").value) || 2;

            const C_target = parseFloat(document.getElementById("targetConc").value) || 1000000;

            const N_seed = parseFloat(document.getElementById("targetSeed").value) || 100000;

 

            const X = sq1 + sq2 + sq3 + sq4;

            const outputDiv = document.getElementById("resultOutput");

            outputDiv.style.display = "block";

 

            if (X === 0) {

                outputDiv.className = "result-text error-text";

                outputDiv.innerHTML = "請輸入細胞計數數值";

                return;

            }

 

            // 1. 計算原濃度 y = (X/4) * DF * 10^4

            const y = (X / 4) * df * 10000;

           

            // 2. 計算總細胞數

            const totalCells = y * V_init;

 

            // 3. 計算目標總體積 (ml)

            const V_total = totalCells / C_target;

 

            // 4. 計算需要額外加入的培養基體積 (ml)

            const V_add = V_total - V_init;

 

            // 5. 計算 Seeding 體積 (μl)

            const V_seed = (N_seed / C_target) * 1000;

 

            if (V_add < 0) {

                outputDiv.className = "result-text error-text";

                outputDiv.innerHTML = `原濃度過低 (${y.toExponential(2)} cells/ml)<br>無法稀釋至 ${C_target.toExponential(2)} cells/ml<br>請重新離心濃縮細胞。`;

            } else {

                outputDiv.className = "result-text";

                outputDiv.innerHTML = `原濃度: ${y.toExponential(2)} cells/ml<br>

                                       請加入 <b>${V_add.toFixed(2)} ml</b> cDMEM<br>

                                       每 well 接種 <b>${V_seed.toFixed(1)} μl</b>`;

               

                saveRecord(name, V_add.toFixed(2), V_seed.toFixed(1));

            }

        }

 

        function saveRecord(name, vAdd, vSeed) {

            const now = new Date();

            const timeStr = `${now.getHours().toString().padStart(2,'0')}:${now.getMinutes().toString().padStart(2,'0')}`;

            const record = { time: timeStr, name: name, add: vAdd, seed: vSeed };

           

            let history = JSON.parse(localStorage.getItem("cellHistory")) || [];

            history.unshift(record);

            if(history.length > 50) history.pop(); // 保留最近50筆

            localStorage.setItem("cellHistory", JSON.stringify(history));

           

            loadHistory();

        }

 

        function loadHistory() {

            let history = JSON.parse(localStorage.getItem("cellHistory")) || [];

            let tbody = document.getElementById("historyBody");

            tbody.innerHTML = "";

            history.forEach(r => {

                let tr = `<tr><td>${r.time}</td><td>${r.name}</td><td>+ ${r.add}</td><td>${r.seed}</td></tr>`;

                tbody.innerHTML += tr;

            });

        }

 

        function clearHistory() {

            if (confirm("確定要清除所有紀錄嗎？(無法復原)")) {

                localStorage.removeItem("cellHistory");

                loadHistory();

                const outputDiv = document.getElementById("resultOutput");

                outputDiv.style.display = "none";

            }

        }
